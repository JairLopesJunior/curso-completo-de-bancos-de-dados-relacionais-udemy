Constraints parte4 - Primary Key Simples

Check - Permite aplicar uma ou várias condições a uma coluna de uma tabela para
permitir ou impedir a persistencia da informação passada para a mesma em uma operação
de inser ou update.

Unique (Aceita valor null) - Garante que o valor de uma coluna ou e um conjunto de colunas será único 
entre todos os registros da tabela.

Primary Key Simples - Garante uma identificação única para cada registro dentro de uma tabela com base em uma coluna.

drop table produto.tb_categoria;
drop table produto.tb_fabricante;
drop table produto.tb_produto;

CREATE TABLE produto.tb_categoria (
    id_categoria serial primary key,
    categoria varchar(30)
);

/* unique e not null */
/* auto_increment / serial */

CREATE TABLE produto.tb_produto (
    id_produto serial primary key,
    fk_id_categoria int,
    fk_id_fabricante int,
    produto varchar(50) not null,
    unidade_medida char(3) not null,
    preco_compra  numeric(10,2) not null,
    preco_venda numeric(10,2) not null
);

CREATE TABLE produto.tb_fabricante (
    id_fabricante serial primary key,
    fabricante varchar(100) not null,
    telefone varchar(17),
    contato varchar(30)
);
